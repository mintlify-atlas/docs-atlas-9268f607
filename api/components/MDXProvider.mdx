## Overview

`MDXProvider` is a context provider that allows you to customize how MDX markdown elements are rendered. It enables you to replace default HTML elements with custom styled components throughout your MDX content.

## Import

```javascript
import { MDXProvider, useMDXComponents, getMDXComponents } from 'ryunix'
```

## Components

### MDXProvider

Context provider for MDX component customization.

```javascript
const { Provider: MDXProvider, useContext: useMDXComponents } = createContext(
  'ryunix.mdx',
  {}
)
```

### useMDXComponents

Hook to access MDX components from context.

```javascript
const useMDXComponents = () => {
  // Returns current MDX components from context
}
```

### getMDXComponents

Utility to merge context components with provided components.

```javascript
const getMDXComponents = (components) => {
  const contextComponents = useMDXComponents()
  return {
    ...contextComponents,
    ...components,
  }
}
```

```jsx Custom Components
import { MDXProvider } from 'ryunix'

const components = {
  h1: (props) => (
    <h1 className="text-4xl font-bold mb-4" {...props} />
  ),
  h2: (props) => (
    <h2 className="text-3xl font-semibold mb-3" {...props} />

  p: (props) => (
    <p className="text-gray-700 mb-4 leading-relaxed" {...props} />
  ),
  a: (props) => (
    <a className="text-blue-600 hover:underline" {...props} />
  ),
  code: (props) => (
    <code className="bg-gray-100 px-2 py-1 rounded" {...props} />
  ),
}

const Blog = ({ children }) => {
  return (
    <MDXProvider value={components}>
      {children}
    </MDXProvider>
  )

```

```jsx Nested Providers
import { MDXProvider, useMDXComponents } from 'ryunix'

const globalComponents = {
  h1: (props) => <h1 className="title" {...props} />,
  p: (props) => <p className="paragraph" {...props} />,
}

const articleComponents = {
  // Override h1, inherit p
  h1: (props) => <h1 className="article-title" {...props} />,
  code: (props) => <code className="code-block" {...props} />,
}

const App = () => {
  return (
    <MDXProvider value={globalComponents}>
      <Layout>
        <MDXProvider value={articleComponents}>
          <Article />
        </MDXProvider>
      </Layout>
    </MDXProvider>
  )
}
```

```jsx Using useMDXComponents Hook
import { useMDXComponents } from 'ryunix'

const CustomMarkdown = ({ content }) => {
  const components = useMDXComponents()
  
  return (
    <div>
      <components.h1>Title</components.h1>
      <components.p>{content}</components.p>
    </div>
  )
}
```

```jsx Merging Components
import { getMDXComponents } from 'ryunix'

const Article = () => {
  // Merge context components with local overrides
  const components = getMDXComponents({
    h1: (props) => <h1 className="custom-h1" {...props} />,
  })
  
  return (
    <article>
      <components.h1>Article Title</components.h1>
      <components.p>Article content...</components.p>
    </article>
  )
}
```

## Advanced Patterns

### Syntax Highlighting

```jsx
import { MDXProvider } from 'ryunix'
import { Prism } from 'prism-react-renderer'

const components = {
  code: ({ children, className }) => {
    const language = className?.replace(/language-/, '') || 'text'
    
    return (
      <Prism
        code={children}
        language={language}
        theme={theme}
      >
        {({ tokens, getLineProps, getTokenProps }) => (
          <pre className={className}>
            {tokens.map((line, i) => (
              <div key={i} {...getLineProps({ line })}>
                {line.map((token, key) => (
                  <span key={key} {...getTokenProps({ token })} />
                ))}
              </div>
            ))}
          </pre>
        )}
      </Prism>
    )
  },
}
```

### Custom Link Component

```jsx
import { MDXProvider } from 'ryunix'

const CustomLink = ({ href, children, ...props }) => {
  const isInternal = href.startsWith('/')
  const isAnchor = href.startsWith('#')
  
  if (isInternal || isAnchor) {
    return <a href={href} {...props}>{children}</a>
  }
  
  return (
    <a 
      href={href} 
      target="_blank" 
      rel="noopener noreferrer"
      {...props}
    >
      {children} ↗
    </a>
  )
}

const components = {
  a: CustomLink,
}
```

### Table of Contents Generation

```jsx
import { MDXProvider, useStore } from 'ryunix'

const TOCProvider = ({ children }) => {
  const [headings, setHeadings] = useStore([])
  
  const components = {
    h2: (props) => {
      const id = props.children.toLowerCase().replace(/\s+/g, '-')
      setHeadings(h => [...h, { id, text: props.children, level: 2 }])
      return <h2 id={id} {...props} />
    },
    h3: (props) => {
      const id = props.children.toLowerCase().replace(/\s+/g, '-')
      setHeadings(h => [...h, { id, text: props.children, level: 3 }])
      return <h3 id={id} {...props} />
    },
  }
  
  return (
    <MDXProvider value={components}>
      <div className="toc">
        <h4>Table of Contents</h4>
        <ul>
          {headings.map(h => (
            <li key={h.id} style={{ marginLeft: (h.level - 2) * 20 }}>
              <a href={`#${h.id}`}>{h.text}</a>
            </li>
          ))}
        </ul>
      </div>
      {children}
    </MDXProvider>
  )
}
```

## Best Practices

<Tip>
  Define your component overrides at the app level for consistency across all MDX content.
</Tip>

<Tip>
  Use nested MDXProviders to create component variations for different sections (e.g., blog vs documentation).
</Tip>

<Note>
  Component overrides receive all props from the original MDX element, so always spread props to preserve functionality.
</Note>

## TypeScript Support

```typescript
import { MDXProvider } from 'ryunix'
import type { ComponentType, HTMLAttributes } from 'ryunix'

type MDXComponents = {
  [key: string]: ComponentType<HTMLAttributes<any>>
}

const components: MDXComponents = {
  h1: (props) => <h1 className="title" {...props} />,
  p: (props) => <p className="paragraph" {...props} />,
}

const App = () => {
  return (
    <MDXProvider value={components}>
      <Content />
    </MDXProvider>
  )
}
```
