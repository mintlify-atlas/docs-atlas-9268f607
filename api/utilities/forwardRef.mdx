## Overview

`forwardRef` allows parent components to pass a `ref` to a child component, enabling direct access to the child's DOM element or component instance. This is essential for imperative actions like focusing inputs or measuring elements.

## Import

```javascript
import { forwardRef } from 'ryunix'
```

## Syntax

```javascript
const forwardRef = (render) => {
  if (typeof render !== 'function') {
    throw new Error('forwardRef requires a render function')
  }

  const ForwardRefComponent = (props) => {
    const { ref, ...restProps } = props || {}
    return render(restProps, ref || null)
  }

  ForwardRefComponent.displayName = `ForwardRef(${render.displayName || render.name || 'Component'})`
  ForwardRefComponent._isForwardRef = true
  ForwardRefComponent._render = render

  return ForwardRefComponent
}
```

## Parameters

<ParamField path="render" type="Function" required>
  A function component that receives `(props, ref)` as arguments. The function should attach the ref to the desired element.
</ParamField>

## Returns

<ResponseField name="ForwardRefComponent" type="Function">
  A component that forwards refs. Marked with `_isForwardRef = true` and includes the original render function in `_render`.
</ResponseField>

## Advanced Patterns

### Combining with useImperativeHandle

```jsx
import { forwardRef, useRef, useImperativeHandle } from 'ryunix'

const FancyInput = forwardRef((props, ref) => {
  const inputRef = useRef(null)
  
  useImperativeHandle(ref, () => ({
    focus: () => {
      inputRef.current.focus()
    },
    clear: () => {
      inputRef.current.value = ''
    },
    getValue: () => {
      return inputRef.current.value
    }
  }))
  
  return <input ref={inputRef} {...props} />
})

const Form = () => {
  const inputRef = useRef(null)
  
  const handleSubmit = () => {
    const value = inputRef.current.getValue()
    console.log('Value:', value)
    inputRef.current.clear()
  }
  
  return (
    <div>
      <FancyInput ref={inputRef} />
      <button onClick={handleSubmit}>Submit</button>
    </div>
  )
}
```

### Multiple Refs

```jsx
import { forwardRef, useRef } from 'ryunix'

const Button = forwardRef((props, ref) => {
  const internalRef = useRef(null)
  
  // Use both internal and forwarded ref
  const setRefs = (element) => {
    internalRef.current = element
    if (ref) {
      if (typeof ref === 'function') {
        ref(element)
      } else {
        ref.current = element
      }
    }
  }
  
  return <button ref={setRefs} {...props} />
})
```

### Conditional Ref Forwarding

```jsx
import { forwardRef } from 'ryunix'

const ConditionalInput = forwardRef((props, ref) => {
  const { type, ...rest } = props
  
  if (type === 'textarea') {
    return <textarea ref={ref} {...rest} />
  }
  
  return <input ref={ref} type={type} {...rest} />
})

const App = () => {
  const inputRef = useRef(null)
  
  return (
    <div>
      <ConditionalInput ref={inputRef} type="text" />
      <ConditionalInput ref={inputRef} type="textarea" />
    </div>
  )
}
```

### Component Library Pattern

```jsx
import { forwardRef } from 'ryunix'

const Button = forwardRef((props, ref) => {
  const { variant = 'primary', size = 'medium', ...rest } = props
  
  const className = `btn btn-${variant} btn-${size}`
  
  return <button ref={ref} className={className} {...rest} />
})

Button.displayName = 'Button'

const Card = forwardRef((props, ref) => {
  return <div ref={ref} className="card" {...props} />
})

Card.displayName = 'Card'

// Usage
const App = () => {
  const buttonRef = useRef(null)
  const cardRef = useRef(null)
  
  return (
    <Card ref={cardRef}>
      <Button ref={buttonRef} variant="primary" size="large">
        Click me
      </Button>
    </Card>
  )
}
```

### Animation Integration

```jsx
import { forwardRef, useRef, useEffect } from 'ryunix'

const AnimatedBox = forwardRef((props, ref) => {
  return <div ref={ref} className="box" {...props} />
})

const App = () => {
  const boxRef = useRef(null)
  
  useEffect(() => {
    if (boxRef.current) {
      // Animate with GSAP, Framer Motion, etc.
      boxRef.current.animate(
        [{ opacity: 0 }, { opacity: 1 }],
        { duration: 1000 }
      )
    }
  }, [])
  
  return <AnimatedBox ref={boxRef}>Animated content</AnimatedBox>
}
```

## Display Name

ForwardRef components automatically get a display name:

```javascript
const Input = forwardRef((props, ref) => {
  return <input ref={ref} {...props} />
})

Input.displayName = 'CustomInput'
// Component appears as ForwardRef(CustomInput) in dev tools

// Or let forwardRef generate it:
function MyInput(props, ref) {
  return <input ref={ref} {...props} />
}

const ForwardedInput = forwardRef(MyInput)
// Component appears as ForwardRef(MyInput)
```

## Common Use Cases

### 1. Form Input Management

```jsx
const Input = forwardRef((props, ref) => {
  return <input ref={ref} {...props} />
})

const LoginForm = () => {
  const emailRef = useRef(null)
  const passwordRef = useRef(null)
  
  const handleSubmit = (e) => {
    e.preventDefault()
    const email = emailRef.current.value
    const password = passwordRef.current.value
    // Submit logic
  }
  
  return (
    <form onSubmit={handleSubmit}>
      <Input ref={emailRef} type="email" />
      <Input ref={passwordRef} type="password" />
      <button type="submit">Login</button>
    </form>
  )
}
```

### 2. Scroll Control

```jsx
const ScrollableList = forwardRef((props, ref) => {

const App = () => {

```

### 3. Canvas Manipulation

```jsx
const Canvas = forwardRef((props, ref) => {
  return <canvas ref={ref} {...props} />
})

const App = () => {

  useEffect(() => {
    const canvas = canvasRef.current
```

## Best Practices

<Tip>
  Always set a `displayName` for better debugging experience in React DevTools.
</Tip>

<Tip>
  Use `forwardRef` when creating reusable component libraries to allow consumers to access underlying DOM elements.
</Tip>

<Note>
  The ref is automatically set to `null` if not provided, so you don't need to check for its existence before using it.
</Note>

<Warning>
  Don't overuse refs for things that can be done declaratively. Use refs for imperative actions like focus, selection, or measurements.
</Warning>

## TypeScript Support

```typescript
import { forwardRef, Ref } from 'ryunix'

interface InputProps {
  label: string
  error?: string
}

const Input = forwardRef<HTMLInputElement, InputProps>(
  ({ label, error, ...props }, ref) => {
    return (
      <div>
        <label>{label}</label>
        <input ref={ref} {...props} />
        {error && <span>{error}</span>}
      </div>
    )
  }
)

// Usage with proper typing
const App = () => {
  const inputRef = useRef<HTMLInputElement>(null)
  
  return <Input ref={inputRef} label="Username" />
}
```
